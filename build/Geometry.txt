sin = function sin(angle) {
    return Math.sin(angle * (Math.PI / 180));
}

cos = function sin(angle) {
    return Math.cos(angle * (Math.PI / 180));
}


function circle(x, y, r) {
	this.x = x || 0;
	this.y = y || 0;
	this.r = r || 1;
	this.color = "none";
	this.stroke = "black";
	this.opacity = 1;
	this.type = 'circle';

	this.scale = function(xTimes) {
		xTimes = xTimes || 1;
		this.r = this.r * xTimes;
	}

}
